31 * @brief Semantically check an argument (e.g. -preset 8)
32 * @param option The argument
33 * @return The token

35std::expected<Token, ErrorType> lex(const std::string& option)
36{
37 auto token_string = option.substr(0, option.find(' '));
38
39 std::string token_value{};
40 if (auto value_pos = option.find_first_not_of(' ', option.find(' ')); value_pos != std::string::npos)//handles flags
41 token_value = option.substr(value_pos, std::string::npos);
42
43
44 TokenType token_type;
45
46 //Find TokenType by checking if token_string is equal to any command
47 if (token_string == token_strings[static_cast<size_t>(TokenType::COMPRESSION_PRESET)]) token_type = TokenType::COMPRESSION_PRESET;
48 else if (token_string == token_strings[static_cast<size_t>(TokenType::FILENAME_IN)]) token_type = TokenType::FILENAME_IN;
49 else if (token_string == token_strings[static_cast<size_t>(TokenType::FILENAME_OUT)]) token_type = TokenType::FILENAME_OUT;
50 else if (token_string == token_strings[static_cast<size_t>(TokenType::N_FILES)]) token_type = TokenType::N_FILES;
51 else if (token_string == token_strings[static_cast<size_t>(TokenType::SIZE_FILES)]) token_type = TokenType::SIZE_FILES;
52 else if (token_string == token_strings[static_cast<size_t>(TokenType::LONG_HELP)]) token_type = TokenType::LONG_HELP;
53 else if (token_string == token_strings[static_cast<size_t>(TokenType::HELP)]) token_type = TokenType::HELP;
54 else if (token_string == token_strings[static_cast<size_t>(TokenType::FORCE_COMPRESSION)]) token_type = TokenType::FORCE_COMPRESSION;
55 else if (token_string == token_strings[static_cast<size_t>(TokenType::DELETE_INPUT)]) token_type = TokenType::DELETE_INPUT;
56 else if (token_string == token_strings[static_cast<size_t>(TokenType::COMPRESSOR_TYPE)]) token_type = TokenType::COMPRESSOR_TYPE;
57 else if (token_string == token_strings[static_cast<size_t>(TokenType::CONCATENATE)]) token_type = TokenType::CONCATENATE;
58 else if (token_string == token_strings[static_cast<size_t>(TokenType::DO_DECOMP_AFTER_CONCAT)]) token_type = TokenType::DO_DECOMP_AFTER_CONCAT;
59 else if (token_string == token_strings[static_cast<size_t>(TokenType::DO_NOT_DECOMP_AFTER_CONCAT)]) token_type = TokenType::DO_NOT_DECOMP_AFTER_CONCAT;
60 else
61 throw_error(ErrorType::SYNTAX_ERROR, token_string);
62
63 //handle flags
64 switch (token_type)
65 {
66 case TokenType::DELETE_INPUT:
67 case TokenType::FORCE_COMPRESSION:
68 case TokenType::CONCATENATE:
69 case TokenType::HELP:
70 case TokenType::LONG_HELP:
71 return Token{ token_type, true };
72 break;
73 default:
74 break;
75 }
76
77
78 ValueType value;
79 //Find token_value and do error checking
80 switch (token_type)
81 {
82 case TokenType::FILENAME_IN:
83 {
84 //Test if we can find and access the input file
85 if (!fs::exists(token_value))
86 throw_error(ErrorType::PATH_NOT_FOUND, token_value);
87
88 if (std::ifstream file(token_value); !file.is_open())
89 throw_error(ErrorType::PATH_NOT_ACCESSIBLE, token_value);
90
91 value.emplace<fs::path>(token_value);
92 }
93 break;
94 case TokenType::FILENAME_OUT:
95 {
96 fs::path path{ token_value };
97
98 //Test if we can write a file to the given dir
99 auto test_path = path.parent_path() / "test_writeability";
100 std::ofstream test_file{test_path};
101 if (!test_file.is_open())
102 {
103 fs::remove(test_path);
104 throw_error(ErrorType::PATH_NOT_ACCESSIBLE, path.string());
105 }
106 test_file.close();
107 fs::remove(test_path);
108
109 path.replace_extension(FILE_EXTENSION);
110
111 value.emplace<fs::path>(path);
112 }
113 break;
114 case TokenType::COMPRESSOR_TYPE:
115 {
116 std::transform(token_value.begin(), token_value.end(), token_value.begin(), [](char c) { return std::toupper(c); });
117
118 for (int i = 0; i < static_cast<int>(CompType::MAX); i++)
119 {
120 if (COMPRESSOR_STR_OPTIONS[i] == token_value)
121 {
122 value.emplace<size_t>(static_cast<size_t>(i));
123 break;
124 }
125 }
126
127 if (std::holds_alternative<std::monostate>(value))
128 throw_error(ErrorType::SYNTAX_ERROR, token_string + token_value);
129 }
130 break;
131 case TokenType::COMPRESSION_PRESET:
132 {
133 //Convert and check the compression preset
134 size_t value_cmpr;
135 auto res = std::from_chars(token_value.data(), token_value.data() + token_value.size(), value_cmpr);
136
137 if (res.ec != std::errc() or (value_cmpr > CompPreset::COMP_MAX or value_cmpr < CompPreset::NO_COMP))//if not a number or outside of our range
138 throw_error(ErrorType::VALUE_ERROR, token_string + token_value);
139
140 value.emplace<size_t>(value_cmpr);
141 }
142 break;
143 case TokenType::N_FILES:
144 {
145 int value_temp;
146 auto res = std::from_chars(token_value.data(), token_value.data() + token_value.size(), value_temp);
147
148 if (res.ec != std::errc() or (value_temp > N_FILES_LIMIT or value_temp < 1))//if less than 1 or over N_FILES_LIMIT
149 throw_error(ErrorType::VALUE_ERROR, token_string + token_value);
150
151 value.emplace<size_t>(value_temp);
152 }
153 break;
154 case TokenType::SIZE_FILES:
155 {
156 int value_temp;
157 auto res = std::from_chars(token_value.data(), token_value.data() + token_value.size(), value_temp);
158
159 if (res.ec != std::errc() or (value_temp < SIZE_FILES_MIN))//if negative or under SIZE_FILES_MIN
160 throw_error(ErrorType::VALUE_ERROR, token_string + token_value);
161
162 value.emplace<size_t>(value_temp);
163 }
164 //the next are all flags handled by the switch just above
165 case TokenType::LONG_HELP:
166 case TokenType::HELP:
167 case TokenType::FORCE_COMPRESSION:
168 case TokenType::DELETE_INPUT:
169 case TokenType::CONCATENATE:
170 case TokenType::DO_DECOMP_AFTER_CONCAT:
171 case TokenType::DO_NOT_DECOMP_AFTER_CONCAT:
172 case TokenType::NO_TYPE:
173 break;
174 }
175
176 return Token{ token_type, value };
177}

179/**
180 * @brief Read the cli.
181 * @param argc Number of cli arguments, where an argument is a 'word' separated by spaces, e.g. -i or -preset
182 * @param argv Pointer to the arguments
183 * @return The options obtained from the cli
184 *
185 * @throws std::runtime_error for any kind of syntactic or semantic error
186 */
187Options parse(int argc, char* argv[])
188{
189 if (argc == 1)
190 throw_error(ErrorType::SYNTAX_ERROR, "No arguments given.");
191
192 Options options;
193 //Parse the cli
194 //starting at 1 means we ignore the first (name of program)
195 for (int index = 1; index < argc; index++)
196 {
197 std::string option{ argv[index] };
198 option += ' ';
199 if (index + 1 < argc)//stay in bounds
200 option += argv[++index];
201
202 //If the path has spaces, this makes sure that we take the whole path (we check for ")
203 //else we throw
204 if (std::ranges::count(option, '\"') == 1)
205 {
206 bool valid = false;
207
208 for(;index < argc - 1; index++)//keep parsing arguments until we find an ending double quote
209 {
210 std::string option_temp{ argv[index + 1] };
211 option += option_temp;
212 if (std::count(option_temp.begin(), option_temp.end(), '\"') == 1)
213 {
214 valid = true;
215 break;
216 }
217 }
218
219 if (!valid)
220 throw_error(ErrorType::PATH_INVALID, option);
221 }
222
223 auto token = lex(option);
224
225 if (token)
226 {
227 switch (token.value().get_type())
228 {
229 case TokenType::FILENAME_IN:
230 options.filename_in = std::get<fs::path>(token.value().get_value());
231 break;
232 case TokenType::FILENAME_OUT:
233 options.filename_out = std::get<fs::path>(token.value().get_value());
234 break;
235 case TokenType::COMPRESSOR_TYPE:
236 options.preset = std::get<size_t>(token.value().get_value());
237 break;
238 case TokenType::COMPRESSION_PRESET:
239 options.preset = std::get<size_t>(token.value().get_value());
240 break;
241 case TokenType::N_FILES:
242 options.n_files = std::get<size_t>(token.value().get_value());
243 break;
244 case TokenType::SIZE_FILES:
245 options.size_files = std::get<size_t>(token.value().get_value());
246 break;
247 //DO index-- FOR FLAGS, it stops us from skipping arguments
248 case TokenType::HELP:
249 case TokenType::LONG_HELP:
250 options.need_help = std::get<bool>(token.value().get_value());
251 if (index + 1 < argc)
252 index--;
253 break;
254 case TokenType::FORCE_COMPRESSION:
255 options.force_compression = std::get<bool>(token.value().get_value());
256 if (index + 1 < argc)
257 index--;
258 break;
259 case TokenType::DELETE_INPUT:
260 options.delete_input = std::get<bool>(token.value().get_value());
261 if (index + 1 < argc)
262 index--;
263 break;
264 case TokenType::CONCATENATE:
265 options.concatenate_files = std::get<bool>(token.value().get_value());
266 if (index + 1 < argc)
267 index--;
268 break;
269 case TokenType::DO_DECOMP_AFTER_CONCAT:
270 options.decomp_after_concat = std::get<bool>(token.value().get_value());
271 if (index + 1 < argc)
272 index--;
273 break;
274 case TokenType::DO_NOT_DECOMP_AFTER_CONCAT:
275 if (index + 1 < argc)
276 index--;
277 break;
278 default:
279 std::terminate();
280 break;
281 }
282
283 }
284 else
285 {
286 throw_error(token.error(), option);
287 }
288 }
289
290 if (options.need_help)
291 std::println(help_str);
292
293 if (fs::is_directory(options.filename_in) and !options.concatenate_files)//tried codecompressing a dir
294 throw_error(ErrorType::DIR_COMPRESSION, options.filename_in.string());
295 else if (options.filename_in.has_filename() and options.concatenate_files)//tried concatenating a file
296 throw_error(ErrorType::PATH_INVALID, options.filename_in.string());
297
298 //swap ifs
299 if (options.filename_in.empty() or options.filename_in.string().find_first_not_of(' ') == std::string::npos)//if filename_in is empty
300 {
301 if (options.concatenate_files)
302 options.filename_in = fs::current_path();
303 else if (!options.need_help)
304 throw_error(ErrorType::MISSING_ARGUMENT, token_strings[static_cast<int>(TokenType::FILENAME_IN)].data());
305 else
306 {
307 std::terminate();//if you need help but no arguments
308 }
309 }
310
311 if (options.filename_out == DEFAULT_OUT_PATH)
312 options.filename_out.replace_extension(FILE_EXTENSION);
313
314 return options;
315}
