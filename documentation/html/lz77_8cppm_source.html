419template <typename DataType>
420void LZ77<DataType>::compress(CodecInterface<Sym, LZ77_Token>& interface) requires IsSym<DataType>
421{
422 interface.out_data.reserve(SIZE_CHUNK);
423
424 size_t local_index = 0;
425 //loop over the stream
426 while (!interface.in_data.reached_end() and local_index < SIZE_CHUNK)
427 {
428 auto token = this->pattern_matcher.find_pattern();
429
430 auto num_of_rolls = static_cast<int>(token.length + 1);
431 this->pattern_matcher.roll_hash(num_of_rolls);//roll, updates the hash and position
432 this->window.slide(num_of_rolls);
433 interface.in_data += num_of_rolls;
434 local_index += num_of_rolls;
435
436 interface.out_data.push_back(token);
437 }
438}

440template<typename DataType>
441void LZ77<DataType>::decompress(CodecInterface<LZ77_Token, Sym>& interface) requires IsToken<DataType>
442{
443 interface.out_data.reserve(interface.in_data.distance_to_end());
444 while (!interface.in_data.reached_end())
445 {
446 if (interface.in_data.iterator->offset != 0)
447 {
448 for (size_t remaining_syms = interface.in_data.iterator->length; remaining_syms > 0; remaining_syms--)
449 interface.out_data.push_back(interface.out_data[interface.out_data.size() - interface.in_data.iterator->offset]);
450 }
451 interface.out_data.emplace_back(interface.in_data.iterator->symbol);
452 interface.in_data++;
453 }
