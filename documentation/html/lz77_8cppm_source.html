406 * @note Will clear out_data buffer
407 */
408 template <typename Allocator = std::allocator<Sym>>
409 void decompress(std::vector<Sym, Allocator>& out_data) const
410 {
411 out_data.reserve(data_.size() / sizeof(Sym));
412 for (auto const token : data_)
413 {
414 if (token.offset != 0)
415 {
416 for (auto _ : std::views::iota(0u, token.length))
417 out_data.push_back(out_data[out_data.size() - token.offset]);
418 }
419 out_data.emplace_back(token.symbol);
420 }

449 * @brief Run the compression

451 void compress() noexcept(false)
452 {
453 auto const partition_size = std::min(data_.size() / concurrency_, MAX_BLOCK_SIZE);
454 for (size_t completed_size = 0, seq_num{}; completed_size < data_.size(); completed_size += partition_size, seq_num++)
455 {
456 //get the true partition size, i.e. the partition can be cut off at the eof

470 block->reserve(data_for_task.size());
471 LZ77Compressor compressor{{reinterpret_cast<Sym const*>(data_for_task.data()), (data_for_task.size_bytes() / sizeof(Sym))}, preset_};
472 compressor.compress(*block);
473 file_list_.insert(std::move(block), seq_num);
474 });
