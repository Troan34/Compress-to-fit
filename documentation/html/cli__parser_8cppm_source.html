159 progress = std::min(1.F, progress);

161
162 std::print("\r\033[K");
163 std::print("\033[34mProgress on the ");
164 if (compressing)
165 std::print("compression:\033[0m");
166 else
167 std::print("extraction:\033[0m");
168
169
170 std::print("[\033[32m");
171 int const filled_width = std::ceil(max_bar_width * progress);
172 for (int i = 0; i < filled_width; i++)
173 {
174 std::print("█");
175 }
176 for (int i = filled_width; i < max_bar_width; i++)
177 {
178 std::print("-");
179 }
180
181
182 std::print("\033[0m] {:.1f}%", progress * 100.F);
183 std::cout << std::flush;
