96 /**
97 * @brief Sequentially apply a certain function to the #cli_options.filename_in file and write the processed data to #out_file

99 * @tparam Out Type of the data written to #out_file

107 {

109
110 size_t offset = 0;
111 bool compressing = true;
112 if constexpr (std::is_same<Out, Sym>::value)
113 {
114 offset = FILE_HEADER_SIZE;
115 compressing = false;
116 }
117
118
119 in.seekg(0, std::ios::end);
120 auto file_size = (static_cast<size_t>(in.tellg()) - offset) / sizeof(In) ;
121 in.seekg(0, std::ios::beg);
122 std::error_code error;
123 auto IO_map = mio::make_mmap_source(cli_options.filename_in.string(), offset, mio::map_entire_file, error);
124
125 //if header is empty and we are decompressing, throw
126 if (!file_options.header.has_value() and std::is_same<Out, Sym>::value)

128 fs::remove(cli_options.filename_out);
129 throw_error(ErrorType::INVALID_DECOMPRESSION, file_options.path.string());
130 }
131
132 std::vector<Out> output_buffer;//Used to receive data from fun()
133
134 CodecInterface<In, Out> codec_interface{
135 .in_data{reinterpret_cast<In const*>(IO_map.data()), reinterpret_cast<In const*>(IO_map.end())},
136 .out_data = output_buffer
137 };
138
139
140 while (!codec_interface.in_data.reached_end())
141 {
142 pred(codec_interface);
143
144 show_progress(cli_options, 1.f - (static_cast<float>(codec_interface.in_data.distance_to_end()) / file_size), compressing);
145 }

170 * @brief Split an encoded file into a certain number of encoded (with header) files
171 * @param path
172 * @param portions
173 */
174 static void split_file(fs::path const& path, size_t portions)
175 {
176
177 if (portions == 1) return;
178
179 if ((portions < 1 or portions > N_FILES_LIMIT) or (fs::file_size(path) / portions) < SIZE_FILES_MIN)
180 print_warn(WarningType::PORTIONS_OUT_OF_RANGE, std::to_string(portions));
181
182
183 std::ifstream source_file{ path, std::ios::binary };
184 assert(has_signature(source_file));
185
186 if (!fs::exists(path)) throw_error(ErrorType::PATH_NOT_FOUND, path.string());
187 auto Header = extract_info(path);
188
189 //move to the ID section
190 source_file.seekg(static_cast<size_t>(HEADER_OFFSET::ID));
191 auto portions_size = fs::file_size(path) / portions;
192
193 std::vector<char> buffer(portions_size);
194
195 auto out_path = path.parent_path() / "output";
196 fs::create_directory(out_path);
197
198
199 //Set up a random number such that num < size_t::max
200 std::random_device seed;
201 std::default_random_engine rng(seed());
202 std::uniform_int_distribution<size_t> dist(1, std::numeric_limits<uint16_t>::max());
203 auto id = dist(rng);
204
205 for (uint16_t file_n = 0; file_n < portions; file_n++)
206 {
207 //create a file with same comp_type and preset, but with id as ID. This file ends with _<file_n>
208 auto file = create_file(Header.value().comp_type, Header.value().preset, out_path / (path.stem().string() + '_' + std::to_string(file_n) + FILE_EXTENSION), id, file_n);
209
210 //We are at the last file, may be smaller than portions_size and/or SIZE_FILES_MIN
211 if (portions_size > (fs::file_size(path) - source_file.tellg()))
212 buffer.resize(fs::file_size(path) - source_file.tellg());
213
214 source_file.read(buffer.data(), buffer.size());
215 file.write(buffer.data(), buffer.size());
216 }
217
218 try
219 {
220 switch (errno)
221 {
222 case ENOSPC:
223 case EIO:
224 case ENFILE:
225 case EMFILE:
226 throw_error(ErrorType::DRIVE_ERROR, path.string());
227 default:
228 break;
229 }
230 }
231 catch (std::runtime_error const& err)
232 {
233 fs::remove_all(out_path);
234 std::terminate();//terminate as to not call fs::remove(path), also, if these errors do happen it is a good idea to drop everything
235 }
236 source_file.close();
237
238 fs::remove(path);
239 }

329 /**
330 * @brief Extract information from the header of the tzf file.
331 * @param path Will not be used to construct a stream
332 * @param file The file the information will be extracted from
333 * @return A FileOptions type constructed with the obtained information
334 *
335 * @pre path shall exist and be accessible.
336 */
337 static std::optional<Header> extract_info(std::ifstream& file, fs::path const& path)
338 {
339
340 //we are dealing with an already compressed file
341 if (has_signature(file))
342 {
343 file.seekg(static_cast<size_t>(HEADER_OFFSET::ID));
344
345 std::byte buffer[sizeof(Header)] = { std::byte{0} };
346 file.read(reinterpret_cast<char*>(buffer), sizeof(Header));
347
348 Header header = std::bit_cast<Header>(buffer);
349
350 CompType comp_type = header.comp_type;
351 CompPreset comp_preset = header.preset;
352
353 if (comp_type >= CompType::MAX or comp_preset > COMP_MAX)
354 throw_error(ErrorType::FILE_CORRUPTED, path.string());
355
356 try_throw_IO_error(file.rdstate(), path);
357
358 return std::optional<Header>{header};
359
360 }
361
362 return std::optional<Header>{};
363 }
