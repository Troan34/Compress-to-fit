125 * @brief Split an encoded file into a certain number of encoded (with header) files
126 * @param path
127 * @param portions
128 */
129 static void split_file(fs::path const& path, size_t portions)
130 {
131
132 if (portions == 1) return;
133
134 if ((portions < 1 or portions > N_FILES_LIMIT) or (fs::file_size(path) / portions) < SIZE_FILES_MIN)
135 print_warn(WarningType::PORTIONS_OUT_OF_RANGE, std::to_string(portions));
136
137
138 std::ifstream source_file{ path, std::ios::binary };
139 assert(has_signature(source_file));
140
141 if (!fs::exists(path)) throw_error(ErrorType::PATH_NOT_FOUND, path.string());
142 auto Header = extract_info(path);
143
144 //move to the ID section
145 source_file.seekg(static_cast<size_t>(HEADER_OFFSET::ID));
146 auto portions_size = fs::file_size(path) / portions;
147
148 std::vector<char> buffer(portions_size);
149
150 auto out_path = path.parent_path() / "output";
151 fs::create_directory(out_path);
152
153
154 //Set up a random number such that num < size_t::max
155 std::random_device seed;
156 std::default_random_engine rng(seed());
157 std::uniform_int_distribution<size_t> dist(1, std::numeric_limits<uint16_t>::max());
158 auto id = dist(rng);
159
160 for (uint16_t file_n = 0; file_n < portions; file_n++)
161 {
162 //create a file with same comp_type and preset, but with id as ID. This file ends with _<file_n>
163 auto file = create_file(Header.value().comp_type, Header.value().preset, out_path / (path.stem().string() + '_' + std::to_string(file_n) + FILE_EXTENSION), id, file_n);
164
165 //We are at the last file, may be smaller than portions_size and/or SIZE_FILES_MIN
166 if (portions_size > (fs::file_size(path) - source_file.tellg()))
167 buffer.resize(fs::file_size(path) - source_file.tellg());
168
169 source_file.read(buffer.data(), buffer.size());
170 file.write(buffer.data(), buffer.size());
171 }
172
173 try
174 {
175 switch (errno)
176 {
177 case ENOSPC:
178 case EIO:
179 case ENFILE:
180 case EMFILE:
181 throw_error(ErrorType::DRIVE_ERROR, path.string());
182 default:
183 break;
184 }
185 }
186 catch (std::runtime_error const&)
187 {
188 fs::remove_all(out_path);
189 std::terminate();//terminate as to not call fs::remove(path), also, if these errors do happen it is a good idea to drop everything
190 }
191 source_file.close();
192
193 fs::remove(path);
194 }

278 /**
279 * @brief Extract information from the header of the tzf file.
280 * @param path Will not be used to construct a stream
281 * @param file The file the information will be extracted from
282 * @return A FileOptions type constructed with the obtained information
283 *
284 * @pre path shall exist and be accessible.
285 */
286 static auto extract_info(std::ifstream& file, fs::path const& path) -> std::optional<Header>
287 {
288
289 //we are dealing with an already compressed file
290 if (has_signature(file))
291 {
292 file.seekg(static_cast<size_t>(HEADER_OFFSET::ID));
293
294 std::byte buffer[sizeof(Header)] = { std::byte{0} };
295 file.read(reinterpret_cast<char*>(buffer), sizeof(Header));
296
297 auto header = std::bit_cast<Header>(buffer);
298
299 if (header.comp_type >= CompType::MAX or header.preset > COMP_MAX)
300 throw_error(ErrorType::FILE_CORRUPTED, path.string());
301
302 try_throw_IO_error(file.rdstate(), path);
303
304 return std::optional{header};
305
306 }
307
308 return std::optional<Header>{};
